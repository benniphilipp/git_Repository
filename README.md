#GIT Repository erstellen
1. cd ordner (in richtiger ordner ebene)
2. git init

#GIT Staus Prüfen
1. git status

#GIT branch erstellen
1. git branch {name}

#GIT Branch Liste
1. git branch --list

#GIT Branches wechseln
1. git checkout {name}

#GIT Hinzufühgen
1. git add .
2. git commit -m 'Text'

#GIT Upload
1. git push --set-upstream origin {name}
